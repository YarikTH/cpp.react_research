Domain.h

Contains the base class for reactive domains and the macro to define them.

Header

#include "react/Domain.h"

Synopsis

Constants

namespace react
{
    // Domain concurrency policies
    enum EDomainMode
    {
        sequential,
        sequential_concurrent,
        parallel,
        parallel_concurrent
    }

    // Optional transaction flags
    enum ETransactionFlags
    {
        allow_merging
    }
}
